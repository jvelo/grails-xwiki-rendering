001package com.monochromeroad.grails.plugins.xwiki;
002
003import org.xwiki.component.manager.ComponentManager;
004import org.xwiki.rendering.syntax.SyntaxFactory;
005import org.xwiki.rendering.transformation.Transformation;
006
007import java.util.List;
008
009public class XWikiRendererConfigurator {
010
011    public static void initialize(XWikiRenderer renderer,
012            XWikiConfigurationProvider configurationProvider, ComponentManager componentManager) {
013        XWikiComponentRepository componentRepository = new XWikiComponentRepository(componentManager);
014
015        renderer.setConfigurationProvider(configurationProvider);
016
017        renderer.setParserFactory(new XWikiParserFactory(componentRepository));
018
019        renderer.setSyntaxFactory(getSyntaxFactory(componentRepository));
020
021        XDOMTransformationManager transformationManager =
022                createTransformationManager(componentRepository, configurationProvider);
023        renderer.setTransformationManager(transformationManager);
024
025        renderer.setWriter(new XDOMWriter(componentRepository));
026    }
027
028    private static XDOMTransformationManager createTransformationManager(
029            XWikiComponentRepository componentRepository, XWikiConfigurationProvider configurationProvider) {
030        XDOMTransformationManager manager = new XDOMTransformationManager();
031        List<String> defaultTransformations = configurationProvider.getDefaultTransformations();
032        for (String transformation : defaultTransformations) {
033            manager.addDefaultTransformation(
034                    componentRepository.lookupComponent(Transformation.class, transformation));
035        }
036        return manager;
037    }
038
039    private static SyntaxFactory getSyntaxFactory(XWikiComponentRepository componentRepository) {
040        return componentRepository.
041                lookupComponent(SyntaxFactory.class);
042    }
043}
