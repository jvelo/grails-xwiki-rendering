001package com.monochromeroad.grails.plugins.xwiki.macro;
002
003import org.apache.commons.lang.StringUtils;
004import org.xwiki.component.descriptor.ComponentDescriptor;
005import org.xwiki.component.descriptor.DefaultComponentDescriptor;
006import org.xwiki.component.phase.InitializationException;
007import org.xwiki.properties.BeanManager;
008import org.xwiki.rendering.block.Block;
009import org.xwiki.rendering.macro.AbstractMacro;
010import org.xwiki.rendering.macro.Macro;
011import org.xwiki.rendering.macro.MacroExecutionException;
012import org.xwiki.rendering.transformation.MacroTransformationContext;
013
014import java.util.List;
015import java.util.Map;
016
017/**
018 * XWiki Macro Runner in a Grails System
019 */
020public class GenericGrailsMacro extends AbstractMacro<Object> {
021
022    public static GenericGrailsMacro getInstance(Map<String, Object> config, BeanManager beanManager) {
023        String macroName = (String)config.get("macroName");
024        String macroFullName = "Grails Macro " + StringUtils.capitalize(macroName);
025        GrailsMacro macroImpl = (GrailsMacro)config.get("macroImpl");
026        Boolean inlineSupport = (Boolean)config.get("inlineSupport");
027        Class parametersBeanClass = (Class)config.get("parametersBeanClass");
028        GenericGrailsMacro macro =
029                new GenericGrailsMacro(macroName, macroFullName,
030                        macroImpl, beanManager, parametersBeanClass, inlineSupport);
031        try {
032            macro.initialize();
033        } catch (InitializationException e) {
034            throw new IllegalStateException(e);
035        }
036
037        return macro;
038    }
039
040    private GrailsMacro macroImpl;
041
042    private String macroName;
043
044    private Boolean inlineSupport = false;
045
046    private GenericGrailsMacro(String macroName, String macroFullName , GrailsMacro macroImpl,
047                               BeanManager beanManager, Class parametersBeanClass, Boolean inlineSupport) {
048        super(macroFullName, macroImpl.toString(), parametersBeanClass);
049        this.macroName = macroName;
050        this.macroImpl = macroImpl;
051        this.beanManager = beanManager;
052        this.inlineSupport = inlineSupport;
053    }
054
055    public boolean supportsInlineMode() {
056        return inlineSupport;
057    }
058
059    public List<Block> execute(Object parameters, String content, MacroTransformationContext context) throws MacroExecutionException {
060        return macroImpl.execute(parameters, content, context);
061    }
062
063    public String getMacroName() {
064        return macroName;
065    }
066
067    public ComponentDescriptor createDescriptor() {
068        DefaultComponentDescriptor<Macro> macroDescriptor =
069                        new DefaultComponentDescriptor<Macro>();
070        macroDescriptor.setImplementation(GenericGrailsMacro.class);
071        macroDescriptor.setRoleType(Macro.class);
072        macroDescriptor.setRoleHint(this.macroName);
073        return macroDescriptor;
074    }
075}
