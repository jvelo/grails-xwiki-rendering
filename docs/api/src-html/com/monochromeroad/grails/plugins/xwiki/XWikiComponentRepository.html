001    package com.monochromeroad.grails.plugins.xwiki;
002    
003    import org.xwiki.component.manager.ComponentLookupException;
004    import org.xwiki.component.manager.ComponentManager;
005    
006    public class XWikiComponentRepository {
007    
008        private final ComponentManager componentManager;
009    
010        public XWikiComponentRepository(ComponentManager componentManager) {
011            this.componentManager = componentManager;
012        }
013    
014        public <T> T lookupComponent(Class<T> componentClass) {
015            try {
016                return componentManager.lookup(componentClass);
017            } catch (ComponentLookupException e) {
018                throw new IllegalStateException(e);
019            }
020        }
021    
022        public <T> T lookupComponent(Class<T> componentClass, String parameter) {
023            try {
024                return componentManager.lookup(componentClass, parameter);
025            } catch (ComponentLookupException e) {
026                throw new IllegalStateException(e);
027            }
028        }
029    }
