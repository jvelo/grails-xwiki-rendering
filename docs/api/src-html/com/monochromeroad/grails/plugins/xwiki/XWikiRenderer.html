001    package com.monochromeroad.grails.plugins.xwiki;
002    
003    import org.xwiki.rendering.block.XDOM;
004    import org.xwiki.rendering.parser.ParseException;
005    import org.xwiki.rendering.parser.Parser;
006    import org.xwiki.rendering.syntax.Syntax;
007    import org.xwiki.rendering.syntax.SyntaxFactory;
008    import org.xwiki.rendering.transformation.Transformation;
009    
010    import java.io.*;
011    
012    /**
013     * Wiki Text Renderer using XWiki Rendering Engine.
014     *
015     * @author Masatoshi Hayashi
016     */
017    public class XWikiRenderer {
018    
019        private XWikiConfigurationProvider configurationProvider;
020    
021        private SyntaxFactory syntaxFactory;
022    
023        private XWikiParserFactory parserFactory;
024    
025        private XDOMBuilder xdomBuilder = new XDOMBuilder();
026    
027        private XDOMTransformationManager xdomTransformationManager;
028    
029        private XDOMWriter xdomWriter;
030    
031        /**
032         * Renders a source text using XWiki Rendering Engine.
033         *
034         * @param source source text reader
035         * @param writer output writer
036         * @param inputSyntax inputSyntax
037         * @param outputSyntax outputSyntax
038         * @param transformations transform parameters
039         */
040        public void render(Reader source, Writer writer,
041                             CharSequence inputSyntax, CharSequence outputSyntax, Transformation ...transformations) {
042            Syntax inputSyntaxObj = getSyntax(inputSyntax);
043            Syntax outputSyntaxObj = getSyntax(outputSyntax);
044    
045            Parser parser = parserFactory.getParser(inputSyntaxObj);
046            XDOM xdom = xdomBuilder.build(source, parser);
047            xdomTransformationManager.transform(xdom, parser, transformations);
048            xdomWriter.write(xdom, outputSyntaxObj, writer);
049        }
050    
051        /**
052         * Renders a source text using XWiki rendering engine and default syntax.
053         *
054         * @param source source text reader
055         * @param writer output writer
056         * @param transformations transform parameters
057         */
058        public void render(Reader source, Writer writer, Transformation ...transformations) {
059            String inputSyntax = configurationProvider.getDefaultInputSyntax();
060            String outputSyntax = configurationProvider.getDefaultOutputSyntax();
061            render(source, writer, inputSyntax, outputSyntax, transformations);
062        }
063    
064        /**
065         * Renders a source text using XWiki rendering engine and default syntax.
066         *
067         * @param source source text reader
068         * @param writer output writer
069         * @param inputSyntax inputSyntax
070         * @param outputSyntax outputSyntax
071         * @param transformations transform parameters
072         */
073        public void render(
074                CharSequence source, Writer writer,
075                CharSequence inputSyntax, CharSequence outputSyntax, Transformation ...transformations) {
076            render(new StringReader(source.toString()), writer, inputSyntax, outputSyntax, transformations);
077        }
078    
079        /**
080         * Renders a source text using XWiki rendering engine and default syntax.
081         *
082         * @param source source text reader
083         * @param writer output writer
084         * @param transformations transform parameters
085         */
086        public void render(CharSequence source, Writer writer, Transformation...transformations) {
087            render(new StringReader(source.toString()), writer, transformations);
088        }
089    
090        /**
091         * Returns rendered text from XWiki Rendering.
092         *
093         * @param source source text reader
094         * @param inputSyntax inputSyntax
095         * @param outputSyntax outputSyntax
096         * @param transformations transform parameters
097         * @return a rendered result
098         */
099        public String render(Reader source, String inputSyntax, String outputSyntax, Transformation ...transformations) {
100            Writer writer = new StringWriter();
101            render(source, writer, inputSyntax, outputSyntax, transformations);
102            return writer.toString();
103        }
104    
105        /**
106         * Returns rendered text from XWiki Rendering using default input syntax and default output syntax.
107         *
108         * @param source a source text
109         * @param transformations transform parameters
110         * @return a rendered result
111         */
112        public String render(Reader source, Transformation ...transformations) {
113            String inputSyntax = configurationProvider.getDefaultInputSyntax();
114            String outputSyntax = configurationProvider.getDefaultOutputSyntax();
115            return render(source, inputSyntax, outputSyntax, transformations);
116        }
117    
118        /**
119         * Returns rendered text from XWiki Rendering.
120         *
121         * @param source a source text
122         * @param inputSyntax inputSyntax
123         * @param outputSyntax outputSyntax
124         * @param transformations transform parameters
125         * @return a rendered result
126         */
127        public String render(CharSequence source,
128                              String inputSyntax, String outputSyntax, Transformation ...transformations) {
129            return render(new StringReader(source.toString()), inputSyntax, outputSyntax, transformations);
130        }
131    
132        /**
133         * Returns rendered text from XWiki Rendering using default input syntax and default output syntax.
134         *
135         * @param source a source text
136         * @param transformations transform parameters
137         * @return a rendered result
138         */
139        public String render(CharSequence source, Transformation ...transformations) {
140            String inputSyntax = configurationProvider.getDefaultInputSyntax();
141            String outputSyntax = configurationProvider.getDefaultOutputSyntax();
142            return render(source, inputSyntax, outputSyntax, transformations);
143        }
144    
145        void setTransformationManager(XDOMTransformationManager transformationManager) {
146            this.xdomTransformationManager = transformationManager;
147        }
148    
149        void setWriter(XDOMWriter writer) {
150            this.xdomWriter = writer;
151        }
152    
153        void setSyntaxFactory(SyntaxFactory syntaxFactory) {
154            this.syntaxFactory = syntaxFactory;
155        }
156    
157        void setParserFactory(XWikiParserFactory parserFactory) {
158            this.parserFactory = parserFactory;
159        }
160    
161        void setConfigurationProvider(XWikiConfigurationProvider configurationProvider) {
162            this.configurationProvider = configurationProvider;
163        }
164    
165        private Syntax getSyntax(Object id) {
166            try {
167                return syntaxFactory.createSyntaxFromIdString(id.toString());
168            } catch (ParseException e) {
169                throw new IllegalArgumentException("Invalid Syntax ID: " + id, e);
170            }
171        }
172    
173    }
